jQuery 快速指南

最流行的 JavaScript 库入门指南

什么是 jQuery？

jQuery 是一个快速、小巧、功能丰富的 JavaScript 库。它使 HTML 文档遍历和操作、事件处理、动画和 Ajax 变得更加简单。

💡 核心理念：写得更少，做得更多 (Write Less, Do More)

基础语法

// 标准写法
$(document).ready(function() {
    // DOM 加载完成后执行的代码
});

// 简写形式
$(function() {
    // DOM 加载完成后执行的代码
});

// 箭头函数写法（现代浏览器）
$(() => {
    // DOM 加载完成后执行的代码
});

1. 选择器（Selector）

jQuery 使用 CSS 选择器语法来选取 HTML 元素。

基本选择器

// 通过 ID 选择
$("#myId")          // 选择 id="myId" 的元素

// 通过类名选择
$(".myClass")       // 选择 class="myClass" 的所有元素

// 通过标签选择
$("p")              // 选择所有 <p> 段落
$("div")            // 选择所有 <div> 元素

// 组合选择
$("div.container")  // 选择 class="container" 的 div

层级选择器

$("ul li")          // 选择 ul 下的所有 li
$("ul > li")       // 选择 ul 的直接子 li
$("h2 + p")         // 选择紧跟在 h2 后的 p
$("div ~ p")        // 选择 div 后的所有同级 p

过滤选择器

$("li:first")       // 第一个 li
$("li:last")        // 最后一个 li
$("li:even")        // 偶数索引的 li (0, 2, 4...)
$("li:odd")         // 奇数索引的 li (1, 3, 5...)
$("li:eq(2)")       // 索引为 2 的 li（第3个）
$("li:gt(1)")       // 索引大于 1 的 li

2. DOM 操作

内容操作

// 获取/设置 HTML 内容
$("#box").html();           // 获取 HTML
$("#box").html("<b>粗体</b>");  // 设置 HTML

// 获取/设置文本内容
$("#box").text();           // 获取纯文本
$("#box").text("普通文本"); // 设置文本（自动转义 HTML）

// 获取/设置表单值
$("#input").val();          // 获取值
$("#input").val("新值");    // 设置值

属性操作

// 获取/设置属性
$("img").attr("src");               // 获取 src 属性
$("img").attr("src", "new.jpg");    // 设置 src 属性
$("img").attr({
    "src": "new.jpg",
    "alt": "新图片"
});                                // 同时设置多个属性

$("img").removeAttr("title");       // 移除属性

样式操作

// 添加/移除类
$("div").addClass("active");        // 添加类
$("div").removeClass("active");     // 移除类
$("div").toggleClass("active");     // 切换类（有则移除，无则添加）
$("div").hasClass("active");        // 检查是否有类（返回 true/false）

// 直接操作样式
$("div").css("color", "red");       // 设置单个样式
$("div").css({                      // 设置多个样式
    "color": "red",
    "font-size": "16px",
    "background-color": "#f0f0f0"
});

$("div").css("color");              // 获取样式值

元素操作

// 添加元素
$("ul").append("<li>新项</li>");       // 在末尾添加
$("ul").prepend("<li>首项</li>");      // 在开头添加
$("li").after("<li>后一项</li>");      // 在某元素后添加
$("li").before("<li>前一项</li>");     // 在某元素前添加

// 删除元素
$("div").remove();                  // 删除元素及子元素
$("div").empty();                   // 清空子元素（保留本身）

// 复制元素
$("div").clone();                   // 克隆元素
$("div").clone(true);               // 克隆并复制事件

3. 事件处理

常用事件

事件 说明 示例
click()点击事件$("#btn").click(fn)
dblclick()双击事件$("#box").dblclick(fn)
mouseenter()鼠标移入$("#box").mouseenter(fn)
mouseleave()鼠标移出$("#box").mouseleave(fn)
focus()获得焦点$("input").focus(fn)
blur()失去焦点$("input").blur(fn)
change()值改变$("select").change(fn)
submit()表单提交$("form").submit(fn)

事件绑定

// 方式 1：直接绑定
$("#btn").click(function() {
    alert("点击了按钮！");
});

// 方式 2：使用 on（推荐，支持动态元素）
$("#btn").on("click", function() {
    alert("点击了按钮！");
});

// 方式 3：事件委托（动态添加的元素也能响应）
$("#list").on("click", "li", function() {
    // 点击 list 中的任意 li 都会触发
    $(this).toggleClass("selected");
});

// 解绑事件
$("#btn").off("click");  // 解绑所有 click 事件

事件对象

$("a").click(function(event) {
    event.preventDefault();  // 阻止默认行为（如跳转）
    event.stopPropagation(); // 阻止事件冒泡
    
    // event 对象常用属性
    event.target;            // 触发事件的元素
    event.currentTarget;     // 当前处理事件的元素（this）
    event.pageX;             // 鼠标点击的 X 坐标
    event.pageY;             // 鼠标点击的 Y 坐标
    event.which;             // 按下的按键（1=左键, 2=中键, 3=右键）
});

4. 动画效果

显示/隐藏

$("#box").show();           // 显示
$("#box").hide();           // 隐藏
$("#box").toggle();         // 切换显示/隐藏

// 带速度参数（毫秒或字符串）
$("#box").show(500);        // 500毫秒
$("#box").hide("slow");     // 慢速（600ms）
$("#box").show("fast");     // 快速（200ms）

// 带回调函数
$("#box").hide(500, function() {
    console.log("动画完成！");
});

淡入淡出

$("#box").fadeIn();         // 淡入
$("#box").fadeOut();        // 淡出
$("#box").fadeToggle();     // 切换淡入/淡出
$("#box").fadeTo(400, 0.5); // 淡到指定透明度（0.5）

滑动

$("#box").slideDown();      // 向下滑动显示
$("#box").slideUp();        // 向上滑动隐藏
$("#box").slideToggle();    // 切换滑动

自定义动画

$("#box").animate({
    width: "300px",
    height: "200px",
    opacity: 0.5,
    marginLeft: "100px"
}, 1000, function() {
    // 动画完成后的回调
    console.log("动画完成");
});

// 链式调用
$("#box")
    .slideUp(300)
    .slideDown(300)
    .fadeOut(300)
    .fadeIn(300);

5. Ajax 请求

无需刷新页面即可与服务器交换数据。

GET 请求

$.get("/api/users", function(data) {
    console.log(data);
});

// 带参数
$.get("/api/users", { id: 1 }, function(data) {
    console.log(data);
});

POST 请求

$.post("/api/users", {
    name: "张三",
    email: "zhangsan@example.com"
}, function(data) {
    console.log("提交成功:", data);
});

通用 AJAX

$.ajax({
    url: "/api/users",
    type: "GET",           // 或 "POST"
    data: { id: 1 },       // 发送的数据
    dataType: "json",      // 期望返回的数据类型
    success: function(data) {
        console.log("成功:", data);
    },
    error: function(xhr, status, error) {
        console.log("错误:", error);
    },
    complete: function() {
        console.log("请求完成");
    }
});

6. 常用方法速查

遍历

$("li").each(function(index, element) {
    console.log(index, $(this).text());
});

// 过滤
$("li").first();    // 第一个
$("li").last();     // 最后一个
$("li").eq(2);      // 第3个（索引2）
$("li").filter(".active");  // 过滤

查找

$("#box").find("p");      // 后代元素
$("#box").children();      // 直接子元素
$("#box").parent();        // 父元素
$("#box").siblings();      // 兄弟元素
$("#box").next();          // 下一个兄弟
$("#box").prev();          // 上一个兄弟

尺寸

$("#box").width();          // 内容宽度
$("#box").height();         // 内容高度
$("#box").innerWidth();     // 含 padding
$("#box").outerWidth();     // 含 padding + border
$("#box").outerWidth(true); // 含 padding + border + margin

位置

$("#box").offset();         // 相对于文档的位置
$("#box").position();       // 相对于父元素的位置
$("#box").scrollTop();      // 垂直滚动位置
$("#box").scrollLeft();     // 水平滚动位置
