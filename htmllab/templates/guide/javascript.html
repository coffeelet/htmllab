JavaScript 快速指南

赋予网页交互能力的神奇脚本语言

变量与数据类型

// 声明变量
let name = "张三";      // 字符串
const age = 25;        // 数字 (常量)
let isStudent = true;  // 布尔值
let skills = ["HTML", "CSS", "JS"]; // 数组

// 输出到控制台
console.log(name);

函数

// 定义函数
function sayHello(userName) {
    return "你好, " + userName + "!";
}

// 调用函数
let message = sayHello("李四");

DOM 操作 (操作 HTML)

// 获取元素
const title = document.getElementById("main-title");

// 修改内容
title.innerText = "新标题";

// 修改样式
title.style.color = "red";

// 添加事件监听
const btn = document.querySelector("button");
btn.addEventListener("click", function() {
    alert("按钮被点击了！");
});

流程控制

// 条件判断
if (age >= 18) {
    console.log("成年人");
} else {
    console.log("未成年人");
}

// 循环
for (let i = 0; i < 5; i++) {
    console.log("次数: " + i);
}
